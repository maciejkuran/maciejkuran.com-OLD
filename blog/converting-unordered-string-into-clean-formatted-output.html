            const flightsInfo = string => {
              const stringToArray = string.split('+');
              
              
              for (let i = 0; i < stringToArray.length; i++) {
              // console.log(stringToArray[i]);
              const [status, from, to, time] = stringToArray[i].split(';');
              //destructuring array and saving values to seperate variables
             
              
              console.log(`${status.includes('Delayed') ? '游댮' : '游릭'} ${status.trim().replaceAll('_', ' ')} from ${from.slice(0, 3).toUpperCase()} to ${to.slice(0, 3).toUpperCase()} (${time.replaceAll(':', 'h')})`);
              }
              };
              
              flightsInfo(flights);
