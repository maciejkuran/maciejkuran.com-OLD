How to make a custom tab component with JavaScript

In this exercise, I am creating a simple tab component that can be useful in any website creation. It will be all done with vanilla JavaScript. The source code is available in the tabs. Please note that I implemented it with JS in 2 different ways. 

                
                    <div class="tab-component-btns-wrapper">
                    <button class="primary-button tab-btn tab-btn--active">HTML</button>
                    <button class="primary-button tab-btn">SCSS</button>
                    <button class="primary-button tab-btn">JS</button>
                  </div>
          
                  <div class="tab-component-content-wrapper">
                    <pre class="tab-content">
                          <code class="language-html">
                              HTML
                          </code>
                      </pre>
                    <pre class="tab-content hide">
                          <code class="language-css">
                              SCSS
                          </code>
                      </pre>
                    <pre class="tab-content hide">
                          <code class="language-js">
                              JS
                          </code>
                      </pre>
                  </div>
                
            
                
                    .tab-component-btns-wrapper {
                        margin: 3rem 0;
                        @include flexCenter();
                      }
                      
                      .tab-component-content-wrapper {
                        position: relative;
                        min-width: 100%;
                        min-height: 40rem;
                      }
                      
                      .tab-content {
                        position: absolute;
                        min-width: 100%;
                        min-height: 100%;
                      }
                      
                      .tab-btn--active {
                        background-color: var(--grey-background-2);
                      }
                
            
                
                    const tabBtns = document.querySelectorAll('.tab-btn');
                    const tabContent = document.querySelectorAll('.tab-content');
                    
                    const switchTabs = e => {
                      let clickedIndex = [...tabBtns].indexOf(e.target);
                      tabContent.forEach(tab => tab.classList.add('hide'));
                      tabBtns.forEach(btn => btn.classList.remove('tab-btn--active'));
                    
                      tabContent[clickedIndex].classList.remove('hide');
                      tabBtns[clickedIndex].classList.add('tab-btn--active');
                    };
                    
                    tabBtns.forEach(btn => btn.addEventListener('click', switchTabs));
