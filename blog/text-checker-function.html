This function checks if text, e.g. submitted by a user includes forbidden words or characters.

If yes, a specific notification will be returned to a user.

              
const forbiddenWords = ['email', '@', 'tel'];
let notification = '';

const textChecker = (text, arr) => {
const input = text.toLowerCase();
if (arr.some(word => input.includes(word))) {
notification = `Forbidden words: ${arr
.join(' ')
.replaceAll(' ', ', ')}`;
} else {
notification = 'Successfully submitted';
}
return notification;
};
              
          
              
const string1 = 'Hi there, How are you ?';
console.log(textChecker(string1, forbiddenWords));
//OUTPUT: Successfully submitted
const string2 = 'Hi ! This is my email adddress: something@gmail.com';
console.log(textChecker(string2, forbiddenWords));
//OUTPUT: Forbidden words: email, @, tel
